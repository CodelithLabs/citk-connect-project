import 'package:google_generative_ai/google_generative_ai.dart';

class GeminiService {
  late final GenerativeModel _model;
  
  GeminiService(String apiKey) {
    _model = GenerativeModel(
      model: 'gemini-1.5-flash',
      apiKey: apiKey,
      systemInstruction: Content.system('''
        You are CITK Bot, the official AI assistant for Central Institute 
        of Technology, Kokrajhar. You have comprehensive knowledge about:
        
        1. Campus Facilities:
           - Library: Open 8 AM - 10 PM (weekdays), 8 AM - 8 PM (weekends)
           - Canteen: Breakfast 7-9 AM, Lunch 12-2 PM, Dinner 7-9 PM
           - Gym: Open 6 AM - 9 PM
        
        2. Academic Information:
           - Semester exams: December & May
           - Internal assessments: Mid-semester
           - Project submissions: End of semester
        
        3. Hostel Rules:
           - Check-in: Before 9 PM
           - No outside guests after 9 PM
           - Quiet hours: 10 PM - 6 AM
        
        4. Transportation:
           - Bus timings: 7 AM, 9 AM, 5 PM from Santinagar
           - Journey time: 30-40 minutes
        
        Always be helpful, concise, and friendly. If asked something you 
        don't know, admit it and suggest contacting the admin office.
      '''),
    );
  }
  
  Future<String> sendMessage(String prompt) async {
    try {
      final response = await _model.generateContent([
        Content.text(prompt),
      ]);
      return response.text ?? 'Sorry, I could not generate a response.';
    } catch (e) {
      return 'Error: Unable to connect to AI service. Please try again.';
    }
  }
  
  Stream<String> sendMessageStream(String prompt) async* {
    try {
      final response = _model.generateContentStream([
        Content.text(prompt),
      ]);
      
      await for (final chunk in response) {
        yield chunk.text ?? '';
      }
    } catch (e) {
      yield 'Error: $e';
    }
  }
}
